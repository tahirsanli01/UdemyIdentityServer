<script>
const values = [
    { value: 40, color: "#ff6b6b", label: "40 Puan" },
    { value: 25, color: "#4dabf7", label: "25 Puan" },
    { value: 20, color: "#51cf66", label: "20 Puan" },
    { value: 15, color: "#ffd43b", label: "15 Puan" }
];

const svg = document.getElementById("chart");

let cumulative = 0;

// Donut arka plan
svg.innerHTML += `
    <circle r="15.915" cx="21" cy="21" fill="transparent" stroke="#eee" stroke-width="6"></circle>
`;

values.forEach(v => {
    const dash = v.value;
    const gap = 100 - dash;
    const offset = -cumulative;

    // Dilimi çiz
    svg.innerHTML += `
        <circle 
            r="15.915" cx="21" cy="21"
            fill="transparent"
            stroke="${v.color}"
            stroke-width="6"
            stroke-dasharray="${dash} ${gap}"
            stroke-dashoffset="${offset}"
        ></circle>
    `;

    // Etiket açısı
    const angle = (cumulative + v.value / 2) * 3.6;
    const rad = (angle - 90) * (Math.PI / 180);

    // Etiket mesafesi – donut'tan tamamen ayrı
    const distance = 115;

    const cx = 160 + Math.cos(rad) * distance;
    const cy = 160 + Math.sin(rad) * distance;

    const div = document.createElement("div");
    div.className = "label";
    div.style.left = cx + "px";
    div.style.top = cy + "px";
    div.innerText = v.label;

    document.querySelector(".chart-container").appendChild(div);

    cumulative += v.value;
});
</script>
