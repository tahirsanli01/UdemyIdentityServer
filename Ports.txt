<script>
const data = [
  {label:"Koyu", value:350, color:"#0b2b4a"},
  {label:"Mavi", value:110, color:"#3b9cff"},
  {label:"Yeşil", value:50,  color:"#2ca24b"},
  {label:"Sarı", value:60,  color:"#f0b229"},
  {label:"Kahverengi", value:90, color:"#c35b43"},
];

const total = data.reduce((a,b)=>a+b.value,0);
const radius = 120;
const stroke = 46;

const segG = document.getElementById("segments");
const lineG = document.getElementById("lines");
const legend = document.getElementById("legend");

let offset = 0;

// Sağdaki etiketlere ZORLAMALI dikey offset (çakışmayı engeller)
const labelOffsets = [0, 26, 52, 78]; 

data.slice(1).forEach((d,i)=>{
  const div=document.createElement("div");
  div.className="item";
  div.innerHTML=`<div class="sw" style="background:${d.color}">${d.value}</div> <span>${d.label}</span>`;
  legend.appendChild(div);
});

// Segmentler
data.forEach((d,i)=>{
  const frac = d.value / total;
  const dash = frac * 2*Math.PI*radius;

  const c = document.createElementNS("http://www.w3.org/2000/svg","circle");
  c.setAttribute("r", radius);
  c.setAttribute("fill","none");
  c.setAttribute("stroke", d.color);
  c.setAttribute("stroke-width", stroke);
  c.setAttribute("stroke-dasharray", `${dash} ${9999}`);
  c.setAttribute("stroke-dashoffset", -(offset * 2*Math.PI*radius));
  segG.appendChild(c);

  // connector çizgisi (ilk büyük olan hariç)
  if(i > 0){
    const midAngle = (offset + frac/2) * 2*Math.PI; 
    const sx = Math.cos(midAngle) * (radius + stroke/2);
    const sy = Math.sin(midAngle) * (radius + stroke/2);

    // sağdaki etiketin SVG koordinatı
    const lx = 170;  
    const ly = -80 + labelOffsets[i-1];

    const line = document.createElementNS("http://www.w3.org/2000/svg","polyline");
    line.setAttribute("class","conn");
    line.setAttribute("points", `${sx},${sy}  ${(sx+lx)/2},${sy}  ${lx},${ly}`);
    line.setAttribute("stroke", data[i].color);
    lineG.appendChild(line);
  }

  offset += frac;
});
</script>
