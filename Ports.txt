<script>
    // Donut merkez koordinatları
    const cx = 250, cy = 250, R = 180;  

    // Her dilimin orta açısı (örnek olarak)
    const angles = [
        45,   // Dilim 1
        135,  // Dilim 2
        225,  // Dilim 3
        315   // Dilim 4
    ];

    // Kutuları yerleştir
    angles.forEach((angle, i) => {
        const rad = (angle - 90) * Math.PI / 180;
        const x = cx + R * Math.cos(rad);
        const y = cy + R * Math.sin(rad);

        const box = document.getElementById("l" + (i+1));
        box.style.left = x + "px";
        box.style.top = y + "px";
    });
</script>
