<canvas id="donut" width="400" height="400"></canvas>

<script>
const canvas = document.getElementById('donut');
const ctx = canvas.getContext('2d');

const data = [
  { label: 'Elma', value: 30, color: '#FF6384' },
  { label: 'Armut', value: 20, color: '#36A2EB' },
  { label: 'Muz', value: 25, color: '#FFCE56' },
  { label: 'Çilek', value: 25, color: '#4BC0C0' }
];

const total = data.reduce((sum, item) => sum + item.value, 0);
const centerX = canvas.width / 2;
const centerY = canvas.height / 2;
const radius = 120;
const innerRadius = 60;
let animationProgress = 0;
let hoverIndex = -1;

function drawChart() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  let startAngle = -Math.PI / 2;

  data.forEach((item, index) => {
    const sliceAngle = (item.value / total) * 2 * Math.PI * animationProgress;

    // Hover efekti
    const midAngle = startAngle + sliceAngle / 2;
    let offsetX = 0;
    let offsetY = 0;
    if (index === hoverIndex) {
      const offset = 15; // ne kadar dışarı çıkacak
      offsetX = offset * Math.cos(midAngle);
      offsetY = offset * Math.sin(midAngle);
    }

    // Dilimi çiz
    ctx.beginPath();
    ctx.arc(centerX + offsetX, centerY + offsetY, radius, startAngle, startAngle + sliceAngle);
    ctx.arc(centerX + offsetX, centerY + offsetY, innerRadius, startAngle + sliceAngle, startAngle, true);
    ctx.closePath();
    ctx.fillStyle = item.color;
    ctx.fill();

    // Etiket
    if (sliceAngle > 0.1) {
      const labelRadius = radius + 20;
      const labelX = centerX + offsetX + labelRadius * Math.cos(midAngle);
      const labelY = centerY + offsetY + labelRadius * Math.sin(midAngle);

      ctx.fillStyle = '#000';
      ctx.textAlign = midAngle > Math.PI / 2 && midAngle < 3 * Math.PI / 2 ? 'right' : 'left';
      ctx.textBaseline = 'middle';
      ctx.fillText(`${item.label} (${item.value})`, labelX, labelY);
    }

    startAngle += sliceAngle;
  });

  // Ortadaki boşluk
  ctx.beginPath();
  ctx.arc(centerX, centerY, innerRadius, 0, 2 * Math.PI);
  ctx.fillStyle = '#f0f0f0';
  ctx.fill();
}

function animate() {
  animationProgress += 0.02;
  if (animationProgress > 1) animationProgress = 1;
  drawChart();
  if (animationProgress < 1) requestAnimationFrame(animate);
}

// Hover tespiti
canvas.addEventListener('mousemove', (e) => {
  const rect = canvas.getBoundingClientRect();
  const x = e.clientX - rect.left - centerX;
  const y = e.clientY - rect.top - centerY;
  const angle = Math.atan2(y, x);
  const distance = Math.sqrt(x * x + y * y);

  if (distance < innerRadius || distance > radius) {
    hoverIndex = -1;
  } else {
    let startAngle = -Math.PI / 2;
    hoverIndex = -1;
    data.forEach((item, index) => {
      const sliceAngle = (item.value / total) * 2 * Math.PI;
      if (angle >= startAngle && angle < startAngle + sliceAngle) {
        hoverIndex = index;
      } else if (angle < -Math.PI / 2) { // açı -PI/2 ile 0 arasında wrap kontrol
        if (angle + 2 * Math.PI >= startAngle && angle + 2 * Math.PI < startAngle + sliceAngle) {
          hoverIndex = index;
        }
      }
      startAngle += sliceAngle;
    });
  }
  drawChart();
});

canvas.addEventListener('mouseleave', () => {
  hoverIndex = -1;
  drawChart();
});

animate();
</script>
